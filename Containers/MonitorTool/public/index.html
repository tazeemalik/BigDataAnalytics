  <style>

    body { font-family: Arial, sans-serif; margin: 16px; }

    .row { display:flex; gap:16px; align-items:flex-start; }

    #chart { width: 70%; }

<body>

  <h1>MonitorTool - cljDetector</h1>

  <div class="row">

    <div id="chart">

      <canvas id="myChart"></canvas>

    </div>

    <div id="status">

      <h3>Status updates</h3>

      <div id="updates"></div>

    </div>

  </div>


  <script>

    async function fetchData() {

      const [samplesRes, statusRes] = await Promise.all([

        fetch('/api/samples'),

        fetch('/api/status?limit=30')

      ]);

      const samples = await samplesRes.json();

      const status = await statusRes.json();

      return { samples, status };

    }


    function renderStatus(status) {

      const updates = document.getElementById('updates');

      updates.innerHTML = '';

      status.forEach(s => {

        const d = new Date(s.ts).toISOString();

        const div = document.createElement('div');

        div.style.marginBottom = '8px';

        div.innerHTML = `<b>${d}</b><div>${(s.iso || '')} ${s.message || ''}</div>`;

        updates.appendChild(div);

      });

    }


    async function renderChart() {

      const { samples, status } = await fetchData();

      renderStatus(status);

      if (!samples.length) return;


      const labels = samples.map(s => new Date(s.ts).toLocaleTimeString());

      const files = samples.map(s => s.files);

      const chunks = samples.map(s => s.chunks);

      const candidates = samples.map(s => s.candidates);

      const clones = samples.map(s => s.clones);


      const ctx = document.getElementById('myChart').getContext('2d');

      if (window._chart) window._chart.destroy();

      window._chart = new Chart(ctx, {

        type: 'line',

        data: {

          labels,

          datasets: [

            { label: 'files', data: files, fill:false, tension:0.2 },

            { label: 'chunks', data: chunks, fill:false, tension:0.2 },

            { label: 'candidates', data: candidates, fill:false, tension:0.2 },

            { label: 'clones', data: clones, fill:false, tension:0.2 }

          ]

        },

        options: {

          interaction: { mode: 'index' },

          plugins: { legend: { position: 'top' } },

          scales: {

            y: { beginAtZero: true }

          }

        }

      });

    }


    // refresh every 10s

    renderChart();

    setInterval(renderChart, 10000);

  </script>

</body>
